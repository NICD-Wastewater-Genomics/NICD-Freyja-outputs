import pandas as pd 
# pull barcode file, default or user provided
df = pd.read_csv('usher_barcodes.csv',index_col=0)

ba2 = df.loc['BA.2']
ba2 = ba2[ba2>0]
ba2Muts = list(ba2.index)

parent = df.loc['BA.2.87']
parent = parent[parent>0]
parentMuts = list(parent.index)

target = df.loc['BA.2.87.1']
target = target[target>0]
targetMuts = list(target.index)

ba287specific = [t for t in parentMuts if t not in ba2Muts]
targetSpecific = [t for t in targetMuts if t not in parentMuts]

print('BA.2.87 specific: ', ba287specific)
print('BA.2.87.1 specific: ', targetSpecific)
